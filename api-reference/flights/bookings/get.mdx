<ParamField path="id" type="string" required>

</ParamField>

## Query Parameters

<ParamField query="date" type="string" format="date">
  Flight date
</ParamField>

<ParamField query="status" type="string">
  Filter by booking status
</ParamField>

## Response (200)

<ParamField status="200" type="FlightBookingsResponse">
  <ParamField name="bookings" type="array">
    List of bookings
  </ParamField>
</ParamField>
