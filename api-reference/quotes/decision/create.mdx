</ParamField>

</ParamField>

<ParamField body="comments" type="string">
  Additional comments
</ParamField>

<ParamField status="200" type="object">
  <ParamField name="quote_id" type="string">

  <ParamField name="status" type="string">

  <ParamField name="decision_at" type="string" format="date-time">

  </ParamField>
